#include<cstdio>
#include<algorithm>
#include<vector>
#include<climits>

#define MAXN 200010
#define int long long
using namespace std;

inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block

signed main(){
    int N=read(),M=read();
    vector<int>a1(N);
    vector<int>a2(M);
    for(int i=0;i<N;++i){
        a1[i]=read();
    }
    for(int i=0;i<M;++i){
        a2[i]=read();
    }
    sort(a2.begin(),a2.end());
    int minn=INT_MAX;
    for(auto &n:a1){
        auto pos=lower_bound(a2.begin(),a2.end(),n);
        if(pos!=a2.end()){
            minn=min(minn,abs(*pos-n));
        }
        if(pos!=a2.begin()){
            --pos;
            minn=min(minn,abs(*pos-n));
        }
    }
    write(minn);
    putchar('\n');
    return 0;
}
