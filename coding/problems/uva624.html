signed main(){
    int lim,N;
    while(lim=read()){
        N=read();
        vector<int>dat(N,0);
        vector<int>dp(lim+1,0);
        vector<vector<int>>route(N,vector<int>(lim+1,0));
        for(int i=0;i<N;++i){
            dat[i]=read();
        }
        //-input;
        for(int i=0;i<N;++i){
            for(int j=lim;j>=dat[i];--j){
                if(dp[j]<dp[j-dat[i]]+dat[i]){
                    dp[j]=dp[j-dat[i]]+dat[i];
                    route[i][j]=1;
                }
            }
        }
        for(int i=N-1,j=lim;i>=0;--i){
            if(route[i][j]){
