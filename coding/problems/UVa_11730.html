//#pragma GCC target("sse,sse2,sse3,ssse3,sse4,popcnt,abm,mmx,avx,tune=native")
//#pragma comment(linker, "/stack:200000000")
#include <cstdio>
#include <queue>

using namespace std;

inline double read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
struct node{
    int p,v;
};

bool isp[10005];
int pri[168],pt=-1;

signed main(){
    for(int i=2;i<10005;++i){
        if(!isp[i]){pri[++pt]=i;}
        for(int j=0;j<=pt;++j){
            if(i*pri[j]>10005){break;}
            isp[i*pri[j]]=true;
            if(i%pri[j]==0){break;}
        }
    }

    int N,T,cases=0;
    while(true){
        N=read();T=read();
        if(N+T==0){break;}

        bool vis[10005]={false};
        queue<node>Q;
        Q.push({N,0});
        vis[N]=true;
        while(!Q.empty()){
            if(Q.front().p==T){
                printf("Case %d: %d\n",++cases,Q.front().v);
                break;
            }
            for(int i=0;i<=pt && pri[i]<Q.front().p;++i){
                if(Q.front().p%pri[i]==0){
                    if(Q.front().p+pri[i]<=T && !vis[Q.front().p+pri[i]]){
                        vis[Q.front().p+pri[i]]=true;
                        Q.push({Q.front().p+pri[i],Q.front().v+1});
                    }
                }
            }
            Q.pop();
            if(Q.empty()){
                printf("Case %d: -1\n",++cases);
            }
        }
    }
    return 0;
}
