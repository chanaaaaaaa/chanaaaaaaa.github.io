        while(T--){
            string s;

            vector<int>dat;
            int pin=0;
            int num=0;
            while(pin<s.size()){
                if(s[pin]=='+'){
                    if(num!=0){
                        dat.emplace_back(num);
                        num=0;
                    }
                    dat.emplace_back(-1);
                }else if(s[pin]=='*'){
                    if(num!=0){
                        dat.emplace_back(num);
                        num=0;
                    }
                    dat.emplace_back(-2);
                }else{
                    num=num*10+s[pin]-'0';
                }
                ++pin;
            }
            if(num!=0){dat.emplace_back(num);num=0;}

            vector<int>maxx=dat;
            for(int i=0;i<maxx.size();++i){
                if(maxx[i]==-1){
                    maxx[i]=maxx[i-1]+maxx[i+1];
                    maxx.erase(maxx.begin()+i+1);
                    maxx.erase(maxx.begin()+i-1);
                    --i;
                }
            }

                    --i;
                }
