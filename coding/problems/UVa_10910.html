#include <cstdio>
#include <memory.h>
#define int long long
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
int dp[71][71];
signed main(){
    int T;
    while(T=read()){
        while(T--){
            int n=read(),t=read(),p=read();
            memset(dp,0,sizeof(dp));
            for(int i=1;i<=t;++i){
                dp[1][i]=1;
            }
            for(int i=1;i<=n;++i){
                for(int j=p;j<=t;++j){
                    for(int k=p;k<=t;++k){
                        if(k-j>=p){
                            dp[i][k]+=dp[i-1][k-j];
                        }
                    }
                }
            }
            write(dp[n][t]);
            putchar('\n');
        }
    }
    return 0;
}
