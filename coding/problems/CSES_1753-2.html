ull pows[MAXN];
ull prefixhash[MAXN];

inline ull getHash(int l,int r){
    return (prefixhash[r]+MOD-(prefixhash[l-1]*pows[r-l+1])%MOD)%MOD;
}

signed main(){
    icc;
    string text,pattern;
    cin>>text>>pattern;

    int n=text.size();
    int m=pattern.size();

    if(n<m){
        cout<<0<<'\n';
        return 0;
    }

    pows[0]=1;
    for(int i=1;i<=n;++i){
        pows[i]=(pows[i-1]*BASE)%MOD;
    }

    prefixhash[0]=0;
    for(int i=1;i<=n;++i){
        ull x=(unsigned char)text[i-1];
        prefixhash[i]=(prefixhash[i-1]*BASE+x)%MOD;
    }

    ull patternHash=0;
    for(int i=0;i<m;++i){
        ull x=(unsigned char)pattern[i];
        patternHash=(patternHash*BASE+x)%MOD;
    }

    int ans=0;
    for(int i=1;i<=n-m+1;++i){
        if(getHash(i,i+m-1)==patternHash){
            ++ans;
        }
    }

    cout<<ans<<'\n';
    return 0;
}
