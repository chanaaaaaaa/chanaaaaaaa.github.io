int d[8][2]={
    {1,2},
    {1,-2},
    {-1,2},
    {-1,-2},
    {2,1},
    {2,-1},
    {-2,1},
    {-2,-1}
};

char s1[3],s2[3];

inline int bfs(){
    queue<pair<int,int>>Q;
    int mp[8][8]={0};

    mp[s1[0]-'a'][s1[1]-'1']=1;
    Q.push({s1[0]-'a',s1[1]-'1'});
    while(!Q.empty()){
        for(int f=0;f<8;++f){
            if(Q.front().first+d[f][0]>=0 && Q.front().first+d[f][0]<8 && Q.front().second+d[f][1]>=0 && Q.front().second+d[f][1]<8){
                if(Q.front().first+d[f][0]==s2[0]-'a' && Q.front().second+d[f][1]==s2[1]-'1'){
                    return mp[Q.front().first][Q.front().second];
                }
                if(!mp[Q.front().first+d[f][0]][Q.front().second+d[f][1]]){
                    mp[Q.front().first+d[f][0]][Q.front().second+d[f][1]]=mp[Q.front().first][Q.front().second]+1;
                    Q.push({Q.front().first+d[f][0],Q.front().second+d[f][1]});
                }
            }
        }
        Q.pop();
    }
    return -1;
}
