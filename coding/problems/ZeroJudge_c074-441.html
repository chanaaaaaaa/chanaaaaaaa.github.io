inline int read(){
    int x=0,w=1;

        c=getchar();
        if(c=='-'){w=-1;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}

int N,M,dat[15],res[10];
inline void dfs(int dep,int st){
    if(dep==6){
        for(int i=0;i<6;++i){

            write(res[i]);

    for(int i=st;i<N;++i){
        res[dep]=dat[i];
        dfs(dep+1,i+1);
    }
    return;
}
