//-block
vector<int>dfn;
vector<int>low;
vector<vector<int>>grid;
vector<bool>instack;

stack<int>s;
map<string,int>mp;
map<int,string>demp;

int dfstime,cnt;

inline void tarjan(int u){
    dfn[u]=low[u]=++dfstime;
    s.push(u);
    instack[u]=true;

    for(auto &v:grid[u]){
        if(!dfn[v]){
            tarjan(v);
            low[u]=min(low[u],low[v]);
        }else if(instack[v]){
            low[u]=min(low[u],dfn[v]);
        }
    }
    if(low[u]==dfn[u]){
        int a=-1;
        do{
            if(a!=-1){cout<<", ";}
            a=s.top();
            s.pop();
            instack[a]=false;
            cout<<demp[a];
        }while(a!=u);
        cout<<'\n';
    }
}
signed main(){
    int N,M,cases=0;
    while(cin>>N>>M && (N || M)){
        cin.ignore();
        dfn.clear();
        low.clear();
        grid.clear();
        instack.clear();
        mp.clear();
        demp.clear();
        while(!s.empty()){s.pop();}

        dfn.assign(N+1,0);
        low.assign(N+1,0);
        instack.assign(N+1,false);
        grid.assign(N+1,vector<int>());
        dfstime=0;cnt=0;

        for(int i=0;i<M;++i){
            string s1,s2;
            int n1,n2;
            cin>>s1>>s2;
            if(mp[s1]==0){
                mp[s1]=++cnt;
                demp[cnt]=s1;
                n1=cnt;
            }else{n1=mp[s1];}

            if(mp[s2]==0){
                mp[s2]=++cnt;
                demp[cnt]=s2;
                n2=cnt;
            }else{n2=mp[s2];}

            grid[n1].push_back(n2);
        }

        if(cases){cout<<'\n';}
        cout<<"Calling circles for data set "<<++cases<<":\n";
        for(int i=1;i<=cnt;++i){
            if(!dfn[i]){
                tarjan(i);
            }
        }
    }
}
