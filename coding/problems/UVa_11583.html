#include <cstdio>
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block

signed main(){
    int T=read();
    while(T--){
        int N=read(),cnt=0,seg=(1<<27)-1;
        while(N--){
            int tmp=0;

            char c=0;
            while(c>'z'||c<'a'){
                c=getchar();
            }
            while(c>='a'&&c<='z'){
                tmp|=(1<<(c-'a'));
                c=getchar();
            }

            if(seg&tmp){
                seg&=tmp;
            }else{
                seg=tmp;
                ++cnt;
            }
        }
        write(cnt);
        putchar('\n');
    }
    return 0;
}
