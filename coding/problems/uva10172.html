using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c==-1){return -1;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
vector<queue<int>>stations;
vector<int>bus;
int T,N,S,Q,cur,clock;
signed main(){
    int T=read();
    while(T--){
        N=read();S=read();Q=read();
        stations.clear();
        bus.clear();
        stations.assign(N,queue<int>());

        for(int i=0;i<N;++i){
            int C=read();
            while(C--){
                stations[i].push(read()-1);
            }
        }

        cur=0;clock=0;
        while(true){
			while(!bus.empty() && (bus.back()==cur || stations[cur].size()<Q)) {
				if(bus.back()!=cur){stations[cur].push(bus.back());}
				bus.pop_back();
				++clock;
			}
			while(!stations[cur].empty() && bus.size()<S) {
				bus.push_back(stations[cur].front());
				stations[cur].pop();
				++clock;
			}
			bool flag=true;
			for(int i=0;i<N;++i){
				if(!stations[i].empty() || !bus.empty()) {
					cur=++cur%N;
					clock+=2;
					flag=false;
					break;
				}
			}
			if(flag){break;}
		}

		write(clock);
		putchar('\n');
    }
	return 0;
}
