//#pragma GCC target("sse,sse2,sse3,ssse3,sse4,popcnt,abm,mmx,avx,tune=native")
//#pragma comment(linker, "/stack:200000000")
#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
inline bool cmp(int &a,int &b){
    if(a<0 && b<0){
        return -a<-b;
    }else if(a<0){
        return -a<b;
    }else if(b<0){
        return a<-b;
    }else{
        return a<b;
    }
}
signed main(){
    int P=read();
    while(P--){
        int N=read(),m;
        vector<int>dat(N,0);
        for(int i=0;i<N;++i){
            dat[i]=read();
        }
        sort(dat.begin(),dat.end(),cmp);

        int cnt=0;
        bool flag;
        for(int i=0;i<dat.size();++i){
            if(i==0){
                if(dat[i]<0){
                    flag=false;
                }else if(dat[i]>0){
                    flag=true;
                }
                ++cnt;
            }else if((flag && dat[i]<0) || (!flag && dat[i]>0)){
                flag=!flag;
                ++cnt;
            }
        }
        write(cnt);
        putchar('\n');
    }
    return 0;
}
