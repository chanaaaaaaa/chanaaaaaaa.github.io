b967 2016APCS

題目編號：b967 2016APCS

題目連結（VJudge / ZeroJudge）

題目大意

給定 \(N\) 個節點與 \(N-1\) 條邊構成的樹（家族圖譜，每條邊表示親子關係），求樹的直徑，即任意兩節點間的最長路徑長度（邊數）。

題解

使用兩次 DFS：從任意節點（如 0）出發找到最遠節點 \(u\)，再從 \(u\) 出發找到最遠節點 \(v\)，\(u\) 到 \(v\) 的距離即為直徑。以鄰接表建圖，dfs 時記錄深度並維護最遠節點與距離。

時間複雜度

O(N)

程式碼

#pragma GCC optimize("Ofast,unroll-loops,no-stack-protector,fast-math")
#include <bits/stdc++.h>
using namespace std;

vector<vector<int>>dic;
int N,a,b,mi=-1,md=-1;

void dfs(int idx,int father,int depth){
    for(auto i : dic[idx]){
        if(i!=father){
            dfs(i,idx,depth+1);
        }
    }
    if(depth>md){
        md=depth;
        mi=idx;
    }
}

int main(){
    cin.tie(0);
    cout.tie(0);
    ios_base::sync_with_stdio(0);

    cin>>N;
    dic.assign(N,{});
    for(int i=1;i<N;i++){
        cin>>a>>b;
        dic[a].push_back(b);
        dic[b].push_back(a);
    }
    dfs(0,-1,0);
    dfs(mi,-1,0);
    cout<<md<<"\n";
    return 0;
}
