const int MAXN=26;
int st,N,dp[MAXN];
bool vis[MAXN];
vector<vector<pair<int,int>>>road;
inline int dfs(int now){
    if(vis[now]){return dp[now];}
    vis[now]=true;
    dp[now]=0;
    for(auto &[next,v]:road[now]){
        dp[now]=max(dp[now],dfs(next)+v);
    }
    return dp[now];
}
inline void solve(){
    st=recd();
    road.assign(MAXN,vector<pair<int,int>>());
    memset(vis,0,sizeof(vis));
    memset(dp,0,sizeof(dp));

    N=read();
    while(N--){
        road[recd()].push_back({recd(),read()});
    }
    write(dfs(st));
    putchar('\n');
}
