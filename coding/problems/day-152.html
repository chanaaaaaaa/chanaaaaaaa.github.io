int N,M,dat[15],res[10];
inline void dfs(int dep,int st){
    if(dep==6){
        for(int i=0;i<6;++i){
            if(i){putchar(' ');}
            write(res[i]);
        }
        putchar('\n');
        return;
    }
    if(st==N){return;}
    for(int i=st;i<N;++i){
        res[dep]=dat[i];
        dfs(dep+1,i+1);
    }
    return;
}
