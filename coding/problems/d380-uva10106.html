int main(){
    icc;
    string A,B;
    while(cin>>A>>B){
        int Asz=A.size(),
            Bsz=B.size();
        vector<int>Anum(Asz);
        vector<int>Bnum(Bsz);
        for(int i=0;i<Asz;++i){
            Anum[Asz-i-1]=A[i]-'0';
        }
        for(int i=0;i<Bsz;++i){
            Bnum[Bsz-i-1]=B[i]-'0';
        }

        int Msz=Asz+Bsz+1;
        vector<int>res(Msz+1,0);
        for(int i=0;i<Asz;++i){
            for(int j=0;j<Bsz;++j){
                res[i+j]+=Anum[i]*Bnum[j];
            }
        }
        for(int i=0;i<Msz;++i){
            res[i+1]+=res[i]/10;
            res[i]%=10;
        }
        while(res[Msz]==0 && Msz>0){
            --Msz;
        }
        for(int i=Msz;i>=0;--i){
            putchar(res[i]+'0');
        }
        putchar('\n');
    }
    return 0;
}
