#include<stdio.h>
#include<vector>
#include<set>
using namespace std;

//--block
set<int>primes;
bool isprime[65001];
inline void primef(){
    for(int i=2;i<=65000;i++){
        if(!isprime[i]){primes.insert(i);}
        for(auto it=primes.cbegin();it!=primes.cend();it++){
            if(*it*i>65000){break;}
            isprime[*it*i]=true;
            if(i%*it==0){break;}
        }
    }
}
inline int solve(int a,int n,int m){
    if(n==1){return a%m;}

    long long tmp=solve(a,n/2,m);
    tmp=(tmp*tmp)%m;
    if(n%2){tmp=(tmp*a)%m;}
    return tmp;
}
set<int>res;
inline void form(){
    primef();
    for(int i=2;i<=65000;i++){
        if(primes.count(i)){continue;}
        int now;
        for(now=2;now<i;now++){
            printf("%d-%d\n",i,now);
            if(solve(now,i,i)!=now){break;}
        }

        if(now==i){res.insert(i);}
    }
}
