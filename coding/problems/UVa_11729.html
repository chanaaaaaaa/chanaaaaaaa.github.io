#include <cstdio>
#include <algorithm>
using namespace std;

inline double read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
const int MAXN=1e5+5;
struct node{
    int a,b;
}dat[MAXN];
inline bool cmp(node &a,node &b){
    return a.b<b.b;
}
signed main(){
    int N,cases=0;
    while(N=read()){
        for(int i=N-1;i>=0;--i){
            dat[i].a=read();
            dat[i].b=read();
        }
        sort(dat,dat+N,cmp);
        int tmp=0,res=0;
        while(N--){
            tmp+=dat[N].a;
            res=max(res,tmp+dat[N].b);
        }
        printf("Case %d: %d\n",++cases,res);
    }
    return 0;
}
