    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}

const int MAXN=100;
int mp[MAXN+1][MAXN+1],
    ed[MAXN+1];

inline int dfs(int a,int b){
    if(ed[a]){return ed[a];}
    int maxx=0;
    for(int i=1;i<=b;++i){
        if(mp[a][i]){
            maxx=max(maxx,dfs(i,b)+1);
        }
    }
    return ed[a]=maxx;
}

        memset(mp,0,sizeof(mp));

            int M=read();
            for(int j=1;j<=M;++j){
                int P=read();
                mp[i][P]=1;
