//#pragma GCC target("sse,sse2,sse3,ssse3,sse4,popcnt,abm,mmx,avx,tune=native")
//#pragma comment(linker, "/stack:200000000")
#include <iostream>
#include <set>
#define int long long
using namespace std;


//-block
const int MAXN=1040;
bool isprime[MAXN+1];
set<int>primes;
signed main(){
    for(int i=2;i<MAXN;++i){
        if(!isprime[i]){primes.emplace(i);}
        for(auto it=primes.cbegin();it!=primes.cend();++it){
            if(*it*i>MAXN){break;}
            isprime[*it*i]=true;
            if(i%*it==0){break;}
        }
    }
    primes.emplace(1);

    string s;
    while(cin>>s){
        int N=0;
        for(char &c:s){
            if(c>='a'&&c<='z'){
                N+=c-'a'+1;
            }else if(c>='A'&&c<='Z'){
                N+=c-'A'+27;
            }
        }
        if(primes.count(N)){
            cout<<"It is a prime word.\n";
        }else{
            cout<<"It is not a prime word.\n";
        }
    }

    return 0;
}
