using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c>'9' || c<'0'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}

signed main(){
    int N,M;
    while(N=read()){
        M=read();
        vector<vector<int>>grid(N,vector<int>(N,0x7fffffff));
        for(int i=0;i<M;++i){
            int s=read(),e=read(),w=read();
            grid[s][e]=grid[e][s]=min(grid[s][e],w);
        }
        for(int i=0;i<N;++i){
            grid[i][i]=0;
        }
        for(int k=0;k<N;++k){
            for(int i=0;i<N;++i){
                for(int j=0;j<N;++j){
                    grid[i][j]=min(grid[i][j],grid[i][k]+grid[k][j]);
                }
            }
        }
        int maxx=0;
        for(int i=0;i<N;++i){
            for(int j=0;j<N;++j){
                if(grid[i][j]!=0x7fffffff){
                    maxx=max(maxx,grid[i][j]);
                }
            }
        }
        int sp=min(read(),min(read(),read()));
        write(ceil(maxx*1.0/sp));
        putchar('\n');
    }
    return 0;
}
