#include <cstdio>
#include <memory.h>
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
const int MAXN=500;
struct E{
    int v,next;
}e[MAXN*MAXN];
int N,K,head[MAXN],color[MAXN];

inline void ADD(int u,int v){
    e[K].v=v;
    e[K].next=head[u];
    head[u]=K++;
}

inline bool Dye(int u){
    for(int i=head[u],v=e[i].v;i;i=e[i].next,v=e[i].v){
        if(color[u]==color[v]){return false;}
        if(!color[v]){
            color[v]=3-color[u];
            if(!Dye(v)){return false;}
        }
    }
    return true;
}

signed main(){
    while(N=read()){
        K=1;
        memset(head,0,sizeof(head));
        int u,v;
        while(u=read()+1){
            v=read()+1;
            if(u+v==2){break;}
            ADD(u-1,v-1),ADD(v-1,u-1);
        }
        memset(color,0,sizeof(color));
        color[1]=1;

        if(Dye(1)){
            putchar('Y');
            putchar('E');
            putchar('S');
        }else{
            putchar('N');
            putchar('O');
        }
        putchar('\n');
    }
    return 0;
}
