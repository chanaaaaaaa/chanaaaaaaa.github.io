    while(cin>>s){

        cout<<"Case "<<con<<": ";
        if(s=="HELLO"){
            cout<<"ENGLISH";
        }else if(s=="HOLA"){
            cout<<"SPANISH";
        }else if(s=="HALLO"){
            cout<<"GERMAN";
        }else if(s=="BONJOUR"){

        }
        cout<<"\n";
        con++;
    }
