#include <cstdio>
#include <vector>
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
vector<int>F;
inline int FindRoot(int x){
    if(F[x]==x){return x;}
    return F[x]=FindRoot(F[x]);
}
inline bool Union(int a,int b){
    a=FindRoot(a);
    b=FindRoot(b);
    if(a==b){
        return false;
    }else{
        F[b]=a;
        return true;
    }
}
signed main(){
    int N,M,caese=0;
    while(N=read()){
        M=read();
        if(N+M==0){break;}
        F.clear();
        for(int i=0;i<=N;++i){
            F.emplace_back(i);
        }
        int res=N;
        while(M--){
            int a=read(),b=read();
            if(Union(a,b)){--res;}
        }
        putchar('C');
        putchar('a');
        putchar('s');
        putchar('e');
        putchar(' ');
        write(++caese);
        putchar(':');
        putchar(' ');
        write(res);
        putchar('\n');
    }
    return 0;
}
