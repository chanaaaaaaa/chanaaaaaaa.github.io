#include <cstdio>
#define int long long
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}

signed main(){
    int N,T=0,res;
    while(N=read()){
        res=0;
        for(int i=1;i*3<N;++i){
            if(((N-i-1)>>1)>=i+1){
                res+=(N-i*3-1)>>1;
            }
        }
        printf("Case %d: %lld\n",++T,res);
