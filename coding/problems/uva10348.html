bool check;
inline void solve(int dat[],int now,int res){
    if(now==5){
        if(res==23){
            check=true;
        }
        return;
    }
    for(int i=0;i<5;++i){
        if(dat[i]==-1){continue;}
        int tmp=dat[i];
        dat[i]=-1;

        solve(dat,now+1,res+tmp);
        solve(dat,now+1,res-tmp);
        solve(dat,now+1,res*tmp);

        dat[i]=tmp;
    }
    return;
}
