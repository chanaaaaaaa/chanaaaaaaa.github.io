/* I/O Optimize */
const int MAXN=1005;
int T,N;
struct Job{
    int t,s,idx;
} job[MAXN];
inline bool cmp(const Job &a,const Job &b){
    // T_i/S_i < T_j/S_j <=> T_i*S_j < T_j*S_i
    int x=a.t*b.s, y=b.t*a.s;
    if(x!=y) return x<y;
    return a.idx<b.idx;  // 字典序最小
}
inline void solve(){
    T=read();
    while(T--){
        N=read();
        for(int i=0;i<N;++i){
            job[i].t=read();
            job[i].s=read();
            job[i].idx=i+1;
        }
        sort(job,job+N,cmp);
        for(int i=0;i<N;++i){
            if(i) putchar(' ');
            write(job[i].idx);
        }
        putchar('\n');
        if(T) putchar('\n');
    }
}

/* Programing area*/
signed main(){
    solve();
    system("pause");
    return 0;
