inline void write(long long x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}

/* I/O Optimize */
const long long maxx=1e8;
int N;
inline void solve(){
    
    while(N=read()){
        long long ans[150],pos=0;  // 需足夠大：N=300 時約需 100 個 base-10^8 位數
        ans[0]=1;

        for(int i=N+2;i<=N*2;++i){
            long long tmp=0;
            for(int j=0;j<=pos;++j){
                ans[j]=ans[j]*i+tmp;
                tmp=ans[j]/maxx;
                ans[j]%=maxx;
            }
            if(tmp>0){
                ans[++pos]=tmp;
            }
        }

        write(ans[pos]);
        for(int i=pos-1;i>=0;--i){
            printf("%08lld",ans[i]);
        }
        putchar('\n');
    }
}

/* Programing area*/
signed main(){
    solve();

    return 0;
