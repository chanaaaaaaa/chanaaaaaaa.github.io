inline vector<int> add(vector<int>&a,vector<int>&b){
    vector<int>tmp(max(a.size(),b.size())+1,0);
    for(int i=0;i<a.size();++i){
        tmp[i]+=a[i];
    }
    for(int i=0;i<b.size();++i){
        tmp[i]+=b[i];
        tmp[i+1]+=tmp[i]/10;
        tmp[i]%=10;
    }
    while(tmp.back()==0 && tmp.size()>1){
        tmp.pop_back();

}
