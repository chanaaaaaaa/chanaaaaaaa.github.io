int LCS[1001][1001];
string s1,s2;
inline int Flcs(int i,int j){
    if(i==0 || j==0){return 0;}
    if(LCS[i][j]!=-1){return LCS[i][j];}
    if(s1[i]==s2[j]){
        return LCS[i][j]=Flcs(i-1,j-1)+1;
    }else{
        return LCS[i][j]=max(Flcs(i-1,j),Flcs(i,j-1));
    }
}
