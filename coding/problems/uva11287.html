inline int read(){

    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}

inline bool prime_check(int N){
    if(N==2){return true;}
    if(N<2 && !N&1){return false;}
    for(int i=3;i*i<=N;i+=2){
        if(N%i==0){
            return false;
        }
    }
    return true;
}
inline int spmod(int a,int n,int m){
    int res=1;
    while(n){
        if(n&1){
            res*=a;
            res%=m;
        }
        a*=a;
        a%=m;
        n>>=1;
    }
    return res;
}
signed main(){
    int P,A;
    while(P=read()){
        A=read();
        if(P+A==0){break;}

        if(!prime_check(P) && spmod(A,P,P)==A%P){

            putchar('e');
            putchar('s');
        }else{
            putchar('n');
            putchar('o');
