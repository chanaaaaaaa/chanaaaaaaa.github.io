signed main(){
    unordered_map<int,string>demp;
    int N,M,cases=0;
    while(cin>>N>>M &&(N||M)){
        vector<vector<int>>grid(N,vector<int>(N,0x7fffffff));
        for(int i=0;i<N;++i){
            string s;
            cin>>s;
            demp[i]=s;
            grid[i][i]=0;
        }
        for(int i=0;i<M;++i){
            int s,e,w;
            cin>>s>>e>>w;
            grid[s-1][e-1]=w;
            grid[e-1][s-1]=w;
        }

        for(int k=0;k<N;++k){
            for(int i=0;i<N;++i){
                for(int j=0;j<N;++j){
                    //if(grid[i][j]==0x7fffffff){grid[i][j]=grid[i][k]+grid[k][j];}
                    grid[i][j]=min(grid[i][j],grid[i][k]+grid[k][j]);
                }
            }
        }
        int minn_id=0;
        for(int i=0;i<N;++i){
            int sum=0;
            for(int j=0;j<N;++j){
                sum+=grid[i][j];
            }
            grid[i][i]=sum;
            if(grid[i][i] < grid[minn_id][minn_id]){
                minn_id=i;
            }
        }
        cout<<"Case #"<<++cases<<" : "<<demp[minn_id]<<'\n';
    }
    return 0;
}
