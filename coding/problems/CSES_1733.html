int main(){
    icc;
    string s;
    cin>>s;

    int n=s.size();

    vector<int> pi(n,0);
    for(int i=1;i<n;++i){
        int j=pi[i-1];
        while(j>0 && s[i]!=s[j]){
            j=pi[j-1];
        }
        if(s[i]==s[j]) ++j;
        pi[i]=j;
    }

    vector<int> periods;
    int len=n;
    while(len>0){
        len=pi[len-1];
        periods.push_back(n-len);
    }

    for(int i=0;i<periods.size();++i){
        if(i>0) cout<<' ';
        cout<<periods[i];
    }
    cout<<'\n';
