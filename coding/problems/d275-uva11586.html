#pragma comment(linker, "/stack:200000000")

#include<iostream>
#include<sstream>

using namespace std;

signed main() {
    int t;
    cin>>t;
    cin.ignore();
    while(t--){
        string A;
        stringstream AA;
        getline(cin,A);
        AA<<A;

        int check=0,con=0;
        while(AA>>A){
            con++;
            if(A=="MM"){check++;}
            else if(A=="FF"){check--;}
        }
        if(!check && con>1){cout<<"LOOP\n";}
        else{cout<<"NO LOOP\n";}
    }
    return 0;
}
