inline int gcd(int a,int b){
    return !b?a:gcd(b,a%b);
}
inline int lcm(int a,int b){
    return a*b/gcd(a,b);
}
signed main(){
    int N,M;
    while(N=read()){
        M=read();
        if(N+M==0){break;}
        vector<int>dat;
        for(int i=0;i<M;++i){dat.emplace_back(read());}

        int e=1<<M, res=N, bit;
        for(int i=1;i<e;++i){
            int l=1;
            bit=0;
            for(int j=0;j<M;++j){
                if(i&(1<<j)){
                    l=lcm(l,dat[j]);
                    ++bit;
                }
            }
            if(bit&1){
                res-=N/l;
            }else{
                res+=N/l;
            }
        }
        write(res);
        putchar('\n');
    }
	return 0;
