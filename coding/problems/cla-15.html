inline void multi(vector<int>&a,vector<int>&b){
    vector<int>tmp(a.size()+b.size(),0);

    for(int i=0;i<a.size();++i){
        for(int j=0;j<b.size();++j){
            tmp[i+j]+=a[i]*b[j];
        }
    }

    for(int i=0;i<tmp.size();++i){
        if(tmp[i]>=10){
            if(i==tmp.size()-1){
                tmp.emplace_back(tmp[i]/10);
            }else{
                tmp[i+1]+=tmp[i]/10;
            }
            tmp[i]%=10;
        }
    }

    while(tmp.size()>1 && tmp.back()==0){
        tmp.pop_back();
    }
    a=tmp;
}
