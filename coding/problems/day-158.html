inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c==-1){return 0;}
        if(c=='-'){w=-1;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
/* I/O Optimize */
vector<vector<int>>G;
vector<int>dat;
int maxx=0,max_f=0;
inline void dfs(int cur,int fa,int dep){
    if(dep>maxx){
        maxx=dep;
        max_f=cur;
    }

    for(int &n:G[cur]){
        if(n==fa){continue;}
        dfs(n,cur,dep+1);
    }
}
inline void solve(){
    int N=read();
    G.assign(N,vector<int>());
    dat.assign(N,0);
    while(true){
        int a=read()-1;
        if(a==-1){break;}
        int b=read()-1;
        G[a].push_back(b);
        G[b].push_back(a);
    }

    dfs(0,-1,0);
    maxx=0;
    dfs(max_f,-1,0);

    write(maxx);
    putchar('\n');
}

/* Programing area*/
signed main(){
