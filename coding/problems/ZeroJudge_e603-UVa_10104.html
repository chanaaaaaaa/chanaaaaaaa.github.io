using namespace std;

inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
int a,b,x,y,g;
inline int ext_gc(int a,int b,int &x,int &y){
    if(b==0){
        x=1,y=0;
        return g=a;
    }
    int d=ext_gc(b,a%b,y,x);
    y-=a/b*x;

    return d;
}

signed main(){
    while(a=read()){
        b=read();
        ext_gc(a,b,x,y);
        write(x);
        putchar(' ');
        write(y);
        putchar(' ');
        write(g);
        putchar('\n');
    }
    return 0;
}
