signed main(){
    icc;
    string s;
    while(getline(cin,s)){
        bool flag=false;
        for(int i=0;i<s.size();++i){
            if(!isdigit(s[i]) && s[i]!='-'){
                flag=true;
                break;
            }
        }

        int x,y;
        stringstream sxy(s);
        if(flag){
            int cnt=0;
            sxy>>x>>y;
            for(int i=x+(x&1);i<=y;i+=2){
                if(i%4){++cnt;}
            }
            cout<<cnt<<'\n';
        }else{
            sxy>>x;
            if(x==0){
                cout<<"2 2\n";
            }else if(x&1){
                cout<<abs((x+1)/2)<<' '<<abs((x+1)/2-1)<<'\n';
            }else{
                if((x/2)&1){
                    cout<<"Bachelor Number.\n";
                }else{
                    cout<<abs(x/4+1)<<' '<<abs(x/4-1)<<'\n';
                }
            }
        }
    }
	return 0;
}
