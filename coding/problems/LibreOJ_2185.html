//-block
const int MAXN=50000;
int pr[MAXN+5],u[MAXN+5],d[MAXN+5],t[MAXN+5],cnt;
bool bp[MAXN+5];
inline void init(int lim){
    bp[0]=bp[1]=u[1]=d[1]=1;
    for(int i=2;i<=lim;++i){
        if(!bp[i]){
            pr[++cnt]=i;
            u[i]=-1;
            d[i]=2;
            t[i]=1;
        }
        for(int j=1;j<=cnt && i*pr[j]<=lim;++j){
            bp[i*pr[j]]=1;
            if(i%pr[j]==0){
                u[i*pr[j]]=0;
                d[i*pr[j]]=d[i]/(t[i]+1)*(t[i]+2);
                t[i*pr[j]]=t[i]+1;
                break;
            }else{
                u[i*pr[j]]=-u[i];
                d[i*pr[j]]=d[i]<<1;
                t[i*pr[j]]=1;
            }
        }
    }
    for(int i=2;i<=lim;++i){
        u[i]+=u[i-1];
        d[i]+=d[i-1];
    }
}

inline int solve(int n,int m){
    int res=0;
    for(int l=1,r;l<=min(n,m);l=r+1){
        r=min(n/(n/l),m/(m/l));
        res+=d[n/l]*d[m/l]*(u[r]-u[l-1]);
    }
    return res;
}

signed main(){
    init(50000);
    int N=read();
    while(N--){
        write(solve(read(),read()));
        putchar('\n');
    }
	return 0;
}
