//#pragma GCC target("sse,sse2,sse3,ssse3,sse4,popcnt,abm,mmx,avx,tune=native")
//#pragma comment(linker, "/stack:200000000")

#include <cstdio>
#include <memory.h>
#include <queue>

using namespace std;

inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
    return;
}
//-block
const int MAXN=3e5+5;
int vis[MAXN],dat[MAXN],T,N,D;
signed main(){
    T=read();
    while(T--){
        N=read();D=read();
        for(int i=0;i<N;++i){
            dat[i]=read();
        }
        memset(vis,0,sizeof(vis));
        while(D--){
            ++vis[read()-1];
        }

        priority_queue<int>tmp;
        priority_queue<int,vector<int>,greater<int>>PQ;
        for(int i=0;i<N;++i){
            tmp.push(dat[i]);
            PQ.push(tmp.top());
            tmp.pop();
            while(vis[i]--){
                write(PQ.top());
                putchar('\n');
                tmp.push(PQ.top());
                PQ.pop();
            }
        }
        if(T){putchar('\n');}
    }
    return 0;
}
