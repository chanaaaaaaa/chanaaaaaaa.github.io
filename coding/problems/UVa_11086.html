//#pragma comment(linker, "/stack:200000000")
#include <cstdio>
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
const int MAXN=1<<21;
bool isp[MAXN],res[MAXN];
int primes[MAXN],pt=-1;

signed main(){
    for(int i=2;i<=MAXN;++i){
        if(!isp[i]){primes[++pt]=i;}
        for(int j=0;j<=pt && i*primes[j]<=MAXN;++j){
            isp[i*primes[j]]=true;
            if(i%primes[j]==0){break;}
        }
    }
    for(int i=0;i<=pt;++i){
        for(int j=0;j<=pt && primes[i]*primes[j]<MAXN;++j){
            res[primes[i]*primes[j]]=true;
        }
    }
    //-form
    int N;
    while(N=read()){
        int cnt=0;
        while(N--){
            if(res[read()]){
                ++cnt;
            }
        }
        write(cnt);
        putchar('\n');
    }
    return 0;
}
