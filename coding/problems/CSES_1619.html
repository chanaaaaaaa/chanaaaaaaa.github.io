#include<cstdio>
#include<vector>
#include<algorithm>
#include<map>

#define int long long
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x<0){putchar('-');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}

signed main(void){
    int N=read();
    map<int,int>mp;

    for(int i=0;i<N;i++){
        mp[read()]++;
        mp[read()]--;
    }
    int now=0,maxx=0;
    for(auto it=mp.begin();it!=mp.end();it++){
        now+=it->second;
        maxx=max(maxx,now);
    }
    write(maxx);
}
