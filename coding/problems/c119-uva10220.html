inline void write(int x){
    if(x<0){putchar(' ');x=-x;}
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block

signed main(){
    int N,carry;
    vector<int>num;
    while(N=read()){
        num.clear();
        num.emplace_back(1);
        for(int i=1;i<=N;++i){
            carry=0;
            for(int j=0;j<num.size();++j){
                num[j]=num[j]*i+carry;
                carry=num[j]/10;
                num[j]%=10;
            }

            if(carry!=0){num.emplace_back(carry);}
            while(num[num.size()-1]>=10){
                carry=num[num.size()-1]/10;
                num[num.size()-1]%=10;
                num.emplace_back(carry);
            }
        }
        //output
        carry=0;
        for(auto &n:num){
            carry+=n;
        }
        write(carry);
        putchar('\n');
    }
