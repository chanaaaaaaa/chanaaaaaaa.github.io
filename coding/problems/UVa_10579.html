#include <cstdio>

using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}

vector<vector<int>>dat;
inline vector<int> add(vector<int>&a,vector<int>&b){
    vector<int>tmp(max(a.size(),b.size())+1,0);
    for(int i=0;i<a.size();++i){
        tmp[i]+=a[i];
    }
    for(int i=0;i<b.size();++i){
        tmp[i]+=b[i];
        tmp[i+1]+=tmp[i]/10;
        tmp[i]%=10;
    }
    while(tmp.back()==0 && tmp.size()>1){
        tmp.pop_back();
    }
    return tmp;
}
inline void form(){

        dat.push_back(add(dat[i-1],dat[i-2]));

    return;
}
inline void print(int x){
    for(int i=dat[x].size()-1;i>=0;--i){
        putchar(dat[x][i]+'0');
    }
    putchar('\n');
}
