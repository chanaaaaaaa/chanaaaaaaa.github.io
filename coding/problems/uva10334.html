const int MAXN=210;
vector<vector<int>>dat(1001,vector<int>(MAXN+1,0));
inline void write(vector<int>&res){
    int i=MAXN;
    while(i--){
        if(res[i]!=0){break;}
    }
    ++i;
    while(i--){
        putchar(res[i]+'0');
    }
}
inline void form(){
    dat[0][0]=1;
    dat[1][0]=2;
    for(int i=2;i<=1000;++i){
        for(int j=0;j<MAXN;++j){
            dat[i][j]+=dat[i-1][j]+dat[i-2][j];
            dat[i][j+1]+=dat[i][j]/10;
            dat[i][j]%=10;
        }
    }
}
signed main(){
    int N;
    form();
    while(N=read()){
        write(dat[N-1]);
        putchar('\n');
    }
