#include <cstdlib>
using namespace std;

inline int read(){
    int x=0;
    char c=0;
    while(c<'0' || c>'9'){
        c=getchar();
        if(c==-1){return 0;}
    }
    while(c>='0' && c<='9'){
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar();
    }
    return x;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
/* I/O Optimize */
int N,T;
inline void solve(){
    T=read();
    while(T--){
        N=read();

        int h=1;
        while((h*(h+1)/2 < N) || (h*(h+1)/2-N)&1){++h;}

        write(h);
        putchar('\n');
        if(T){putchar('\n');}
    }
}

/* Programing area*/
signed main(){
    solve();
    system("pause");
    return 0;
}
