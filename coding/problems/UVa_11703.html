//#pragma comment(linker, "/stack:200000000")
#include <cstdio>
#include <math.h>
using namespace std;

inline double read(){
    int x=0,w=1;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
const int MAXN=1000000;
int dat[MAXN+1],N;
signed main(){
    dat[0]=1;
    for(int i=1;i<=MAXN;++i){
        dat[i]=(dat[int(i-sqrt(i+0.0))]+dat[int(log(i+0.0))]+dat[int(sin(i)*sin(i)*i)])%MAXN;
    }

    while(true){
        N=read();
        if(N==-1){break;}
        write(dat[N]);
        putchar('\n');
    }
    return 0;
}
