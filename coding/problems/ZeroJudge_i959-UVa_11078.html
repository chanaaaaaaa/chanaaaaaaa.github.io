//#pragma comment(linker, "/stack:200000000")
#include <cstdio>
using namespace std;

inline int read(){
    int x=0,w=1;
    char c=0;
    while(c<'0'||c>'9'){
        c=getchar();
        if(c=='-'){w=-1;}
        if(c==-1){return 0;}
    }
    while(c>='0'&&c<='9'){
        x=(x<<3)+(x<<1)+(c-'0');
        c=getchar();
    }
    return x*w;
}
inline void write(int x){
    if(x>=10){write(x/10);}
    putchar(x%10+'0');
}
//-block
int T,N,last,now,maxx;
signed main(){
    T=read();
    while(T--){
        N=read()-1;last=read();maxx=-300000;
        while(N--){
            now=read();
            if(last-now > maxx){
                maxx=last-now;
            }
            if(last-now < 0){
                last=now;
            }
        }
        if(maxx<0){putchar('-');maxx=-maxx;}
        write(maxx);
        putchar('\n');
    }
    return 0;
}
